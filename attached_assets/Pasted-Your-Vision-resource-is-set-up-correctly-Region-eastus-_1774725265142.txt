```python
from azure.ai.vision.imageanalysis.models import VisualFeatures
from azure.core.credentials import AzureKeyCredential

client = ImageAnalysisClient(
    endpoint="https://azurevision-test132121.cognitiveservices.azure.com/",
    credential=AzureKeyCredential("your-key-here")
)

# Single call — gets everything
result = client.analyze(
    image_data=image_bytes,  # raw bytes of uploaded image
    visual_features=[
        VisualFeatures.CAPTION,
        VisualFeatures.DENSE_CAPTIONS,
        VisualFeatures.TAGS,
        VisualFeatures.OBJECTS,
        VisualFeatures.READ,
    ]
)

# What you get back:
# result.caption.text → "A marketing image with ZONNIC branding on green gradient"
# result.caption.confidence → 0.89

# result.dense_captions.list → [{text: "logo in center", bbox: {x,y,w,h}}, ...]

# result.tags.list → [{name: "logo", confidence: 0.95}, {name: "gradient", confidence: 0.88}]

# result.objects.list → [{tags: [{name: "text"}], bounding_box: {x,y,w,h}}]

# result.read.blocks[0].lines → [{text: "ZONNIC", bounding_polygon: [...]}]
# result.read.blocks[0].lines → [{text: "WARNING:", bounding_polygon: [...]}]
```
